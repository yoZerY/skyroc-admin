# @skyroc/core-i18n

> 国际化核心逻辑 - 跨平台支持 (Web + React Native)

## 📦 包信息

- **包名**: `@skyroc/core-i18n`
- **版本**: `1.0.0`
- **平台**: Universal (Web + React Native)
- **依赖**:
  - `i18next` - 国际化核心
  - `jotai` - 状态管理
  - `@skyroc/core-storage` - 语言持久化
  - `@skyroc/core-types` - 类型定义

## 🎯 职责定位

**核心职责**:
- 语言切换逻辑
- 语言状态管理
- i18next 配置抽象
- 多语言类型安全

**不负责**:
- ❌ react-i18next hooks (由 `@skyroc/adapter-react-i18next` 提供)
- ❌ 翻译文本管理（由应用层管理）
- ❌ 语言检测（由应用层实现）

## 🔨 核心实现

### 1. Lang Atom

```ts
// src/atoms/lang.ts
import { atom } from 'jotai'
import { atomWithStorage } from 'jotai/utils'
import { defaultLangConfig } from '../config/default'
import type { LangType } from '../types'

// 使用 atomWithStorage 自动持久化
export const langAtom = atomWithStorage<LangType>(
  'app-lang',
  defaultLangConfig.defaultLang
)

// 派生 atom: 当前语言选项
export const currentLangOptionAtom = atom((get) => {
  const lang = get(langAtom)
  return defaultLangConfig.langOptions.find(opt => opt.value === lang)
})
```

### 2. useLang Hook

```ts
// src/hooks/use-lang.ts
import { useAtom, useAtomValue } from 'jotai'
import { langAtom, currentLangOptionAtom } from '../atoms/lang'
import { defaultLangConfig } from '../config/default'
import { storage } from '@skyroc/core-storage'
import type { LangType } from '../types'

export function useLang() {
  const [locale, setLocale] = useAtom(langAtom)
  const currentOption = useAtomValue(currentLangOptionAtom)

  function changeLocale(lang: LangType) {
    // 1. 更新状态
    setLocale(lang)

    // 2. 持久化存储
    storage.set('lang', lang)

    // 3. 更新 i18next（如果已集成）
    if (typeof window !== 'undefined' && (window as any).i18n) {
      ;(window as any).i18n.changeLanguage(lang)
    }
  }

  return {
    // 状态
    locale,
    currentOption,

    // 配置
    localeOptions: defaultLangConfig.langOptions,
    fallbackLang: defaultLangConfig.fallbackLang,

    // 操作
    changeLocale,
    setLocale: changeLocale,

    // 工具方法
    isCurrentLang: (lang: LangType) => locale === lang
  }
}
```

### 3. i18next 配置

```ts
// src/config/i18n.ts
import i18n from 'i18next'
import type { I18nConfig, LangType } from '../types'

export function initI18n(config: I18nConfig) {
  return i18n.init({
    lng: config.lng,
    fallbackLng: config.fallbackLng,
    resources: config.resources,

    // 通用配置
    interpolation: {
      escapeValue: false
    },
    react: {
      useSuspense: false
    },

    // 可选配置
    ...config
  })
}

export function setLng(lang: LangType) {
  return i18n.changeLanguage(lang)
}

export function getCurrentLang(): LangType {
  return i18n.language as LangType
}
```

### 5. 工具函数

```ts
// src/utils/helpers.ts
import { storage } from '@skyroc/core-storage'
import type { LangType } from '../types'

export function getLangFromStorage(): LangType | null {
  return storage.get('lang')
}

export function saveLangToStorage(lang: LangType) {
  storage.set('lang', lang)
}

export function getLangLabel(lang: LangType, options: LangOption[]): string {
  const option = options.find(opt => opt.value === lang)
  return option?.label || lang
}
```

## 💡 使用示例

### 示例 1: 基本使用

```tsx
import { useLang } from '@skyroc/core-i18n'

function LanguageSelector() {
  const { locale, localeOptions, changeLocale } = useLang()

  return (
    <select
      value={locale}
      onChange={(e) => changeLocale(e.target.value as LangType)}
    >
      {localeOptions.map(option => (
        <option key={option.value} value={option.value}>
          {option.label}
        </option>
      ))}
    </select>
  )
}
```

### 示例 2: 语言切换按钮

```tsx
import { useLang } from '@skyroc/core-i18n'

function LanguageToggle() {
  const { locale, changeLocale } = useLang()

  const toggleLang = () => {
    const nextLang = locale === 'zh-CN' ? 'en-US' : 'zh-CN'
    changeLocale(nextLang)
  }

  return (
    <button onClick={toggleLang}>
      {locale === 'zh-CN' ? '中文' : 'English'}
    </button>
  )
}
```

### 示例 3: 集成 react-i18next

```tsx
// 配合 @skyroc/adapter-react-i18next 使用
import { useLang } from '@skyroc/core-i18n'
import { useTranslation } from 'react-i18next'

function MyComponent() {
  const { locale, changeLocale } = useLang()
  const { t } = useTranslation()

  useEffect(() => {
    // 同步到 i18next
    changeLocale(locale)
  }, [locale])

  return <div>{t('common.welcome')}</div>
}
```

```ts
// packages/core-i18n/src/__tests__/use-lang.test.ts
import { renderHook, act } from '@testing-library/react'
import { useLang } from '../hooks/use-lang'

describe('useLang', () => {
  it('should initialize with default language', () => {
    const { result } = renderHook(() => useLang())
    expect(result.current.locale).toBe('zh-CN')
  })

  it('should change language', () => {
    const { result } = renderHook(() => useLang())

    act(() => {
      result.current.changeLocale('en-US')
    })

    expect(result.current.locale).toBe('en-US')
  })

  it('should persist language to storage', () => {
    const { result } = renderHook(() => useLang())

    act(() => {
      result.current.changeLocale('ja-JP')
    })

    // 验证 storage 调用
    expect(storage.get('lang')).toBe('ja-JP')
  })
})
```
